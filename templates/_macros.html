{# A small reusable helper to draw filled/empty stars for a rating. #} {% macro stars(avg, count) %} {% set filled = (avg or 0) | round(0, 'floor') | int %} {%- for i in range(1, 6) -%} {%- if i <= filled %}★{% else %}★{% endif -%} {%- endfor -%} {% if count and count > 0 %} {{ avg }} ({{ count }} review{{ 's' if count != 1 }}) {% else %} No ratings yet {% endif %} {% endmacro %}
